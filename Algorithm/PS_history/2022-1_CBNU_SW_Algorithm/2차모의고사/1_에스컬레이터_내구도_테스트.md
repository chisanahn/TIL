## 성공. 약 50분

문제를 이해하는데 시간이 꽤 걸렸다.

<details><summary>소스코드</summary>

```java
import java.io.*;
import java.util.*;

class Main {

    int N, K;
    int[] A = new int[201];
    boolean[] passengers = new boolean[101];
    int start = 0, zeroCnt = 0;

    void rotate() {
        start = (start + 2 * N - 1) % (2 * N);

        for (int i = N - 2; i >= 0; i--) {
            passengers[i + 1] = passengers[i];
        }
        passengers[0] = false;
        passengers[N - 1] = false;
    }

    void update() {
        for (int i = N - 2; i >= 1; i--) {
            if (!passengers[i]) continue;

            int next = (start + i + 1) % (2 * N);
            if (A[next] > 0 && !passengers[i + 1]) {
                A[next]--;
                if(A[next] == 0) zeroCnt++;
                passengers[i + 1] = true;
                passengers[i] = false;
            }
        }
    }

    void add() {
        if (A[start] > 0) {
            A[start]--;
            if(A[start] == 0) zeroCnt++;
            passengers[0] = true;
        }
    }

    void solution() throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        StringTokenizer st = new StringTokenizer(br.readLine());
        N = Integer.parseInt(st.nextToken());
        K = Integer.parseInt(st.nextToken());
        A = Arrays.stream(br.readLine().split(" ")).mapToInt(Integer::parseInt).toArray();

        int step = 1;
        for (; ; step++) {
            rotate();
            update();
            add();
            if (zeroCnt >= K) break;
        }
        System.out.println(step);

        br.close();
    }

    public static void main(String[] args) throws Exception {
        new Main().solution();
    }
}
```
